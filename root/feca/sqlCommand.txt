create table escola(
idEscola int (5) AUTO_INCREMENT PRIMARY KEY,
cidade varchar(40),
    contatoNumero char (12),
    contatoEmail varchar(70)
);

create table convidados (
idConv int (5) AUTO_INCREMENT PRIMARY KEY,
cpf char (14),
rg char(12),
nome varchar(50) not null,
voto binary,
senha varchar(50) not null, 
escolaID int(5),
    FOREIGN KEY (escolaID) REFERENCES escola(idEscola)
);

create table banda(
idBanda int(5) AUTO_INCREMENT PRIMARY KEY,
    numeroDeIntegrantes decimal(2),
    horarioData timestamp,
    contatoBanda char(12),
    idEscola int(5),
    FOREIGN KEY (idEscola) REFERENCES escola(idEscola)
)
