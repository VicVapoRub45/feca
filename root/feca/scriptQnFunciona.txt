<script>
        function fetchEscolas(query) {
            // Obter a cidade selecionada para enviar junto na requisição
            var cidade = document.getElementById("cidadeSelect").value;

            if (query.length > 2) { // Buscar se o usuário tiver digitado pelo menos 3 caracteres
                var xhr = new XMLHttpRequest();
                xhr.onreadystatechange = function() {
                    if (this.readyState == 4 && this.status == 200) {
                        document.getElementById("sugestoesEscolas").innerHTML = this.responseText;
                    }
                };
                xhr.open("GET", "php/getEscolas.php?query=" + query + "&cidade=" + cidade, true);
                xhr.send();
            } else {
                document.getElementById("sugestoesEscolas").innerHTML = "";
            }
        }

        function selecionarEscola(nome) {
            document.getElementById("escolaInput").value = nome;
            document.getElementById("sugestoesEscolas").innerHTML = ""; // Limpar a lista de sugestões
        }
    </script>
